  title: string;
  subtitle: string;

}

  title,
  subtitle,
  primaryCtaText,
  primaryCtaLink,
  secondaryCtaText,
  secondaryCtaLink,
}: PageHeaderProps) {
  return (
    <div className="bg-dark-navy text-white py-20 px-4 sm:px-6 lg:px-8">
      <div className="max-w-4xl mx-auto text-center">
        <h1 className="text-4xl sm:text-5xl font-bold mb-6 leading-tight">
          {title}
        </h1>
        <p className="text-xl text-gray-300 mb-10 max-w-3xl mx-auto font-light leading-relaxed">
          {subtitle}
        </p>
        <div className="flex flex-col sm:flex-row gap-4 justify-center items-center">
          {primaryCtaText && primaryCtaLink && (
            <Link href={primaryCtaLink}>
              <button className="bg-accent-gold text-dark-navy px-8 py-3 rounded-lg font-semibold hover:-translate-y-0.5 transition-all duration-300 shadow-lg flex items-center">
                {primaryCtaText} <ArrowRight className="ml-2 h-4 w-4" />
              </button>
            </Link>
          )}
          {secondaryCtaText && secondaryCtaLink && (
            <Link href={secondaryCtaLink}>
              <button className="border-2 border-white text-white px-8 py-3 rounded-lg font-semibold hover:bg-white hover:text-dark-navy transition-all duration-300">

    </div>
  );

// client/src/components/navigation.tsx
import { useState, useEffect } from "react";
import { Link, useLocation } from "wouter"; // Updated to use Link and location
import { Menu, X } from "lucide-react";

export default function Navigation() {
  const [isMenuOpen, setIsMenuOpen] = useState(false);
  const [location, setLocation] = useLocation();

  const toggleMenu = () => {
    setIsMenuOpen(!isMenuOpen);
  };

  const handleNavClick = (sectionId: string) => {
    // If we are on home page, scroll to section
    if (location === "/") {
      const element = document.getElementById(sectionId);
      if (element) {
        element.scrollIntoView({ behavior: 'smooth', block: 'start' });
      }
    } else {
      // If not on home page, navigate to home then scroll (simple nav for now just goes to home)
      setLocation(`/?section=${sectionId}`); 
      // Note: Full hash navigation implementation would require handling the hash on home page load
      // For this implementation, we'll simple redirect to home for "anchor" links
      window.location.href = `/#${sectionId}`;
    }
    setIsMenuOpen(false);
  };

  useEffect(() => {
    const handleResize = () => {
      if (window.innerWidth >= 768) {
        setIsMenuOpen(false);
      }
    };

    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  const NavLink = ({ target, children, isPage = false }: { target: string, children: React.ReactNode, isPage?: boolean }) => {
    if (isPage) {
      return (
        <Link href={target}>
          <a className="text-gray-700 hover:text-primary-blue transition-colors duration-300 font-medium">
            {children}
          </a>
        </Link>
      );
    }
    return (
      <button 
        onClick={() => handleNavClick(target)} 
        className="text-gray-700 hover:text-primary-blue transition-colors duration-300 font-medium"
      >
        {children}
      </button>
    );
  };

  const MobileNavLink = ({ target, children, isPage = false }: { target: string, children: React.ReactNode, isPage?: boolean }) => {
    if (isPage) {
      return (
        <Link href={target}>
          <a className="block w-full text-left text-gray-700 hover:text-primary-blue transition-colors duration-300 py-2">
            {children}
          </a>
        </Link>
      );
    }
    return (
      <button 
        onClick={() => handleNavClick(target)} 
        className="block w-full text-left text-gray-700 hover:text-primary-blue transition-colors duration-300 py-2"
      >
        {children}
      </button>
    );
  };

  return (
    <nav className="fixed top-0 left-0 right-0 z-50 bg-white/95 backdrop-blur-md border-b border-gray-200 shadow-sm">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
        <div className="flex justify-between items-center h-20">
          <div className="flex items-center">
            <Link href="/">
              <div className="text-2xl font-bold text-primary-blue cursor-pointer flex items-center gap-2">
                <span className="text-accent-gold text-3xl">✦</span> Alliance Enterprise
              </div>
            </Link>
          </div>
          
          {/* Desktop Navigation */}
          <div className="hidden md:block">
            <div className="flex items-center space-x-8">
              <NavLink target="vision">Vision</NavLink>
              <NavLink target="gallery">Gallery</NavLink>
              <NavLink target="services">Services</NavLink>
              <NavLink target="/for-beginners" isPage>Beginners Hub</NavLink>
              <NavLink target="community">Community</NavLink>
              <button 
                onClick={() => handleNavClick('contact')} 
                className="bg-primary-blue text-white px-6 py-2.5 rounded-lg hover:-translate-y-0.5 transition-all duration-300 shadow-md hover:shadow-lg font-medium"
              >
                Get Started
              </button>
            </div>
          </div>

          {/* Mobile menu button */}
          <div className="md:hidden">
            <button 
              onClick={toggleMenu}
              className="text-gray-700 hover:text-primary-blue p-2"
            >
              {isMenuOpen ? <X className="h-6 w-6" /> : <Menu className="h-6 w-6" />}
            </button>
          </div>
        </div>
      </div>

      {/* Mobile Navigation */}
      {isMenuOpen && (
        <div className="md:hidden bg-white border-t border-gray-200 absolute w-full shadow-lg">
          <div className="px-4 py-4 space-y-2">
            <MobileNavLink target="vision">Vision</MobileNavLink>
            <MobileNavLink target="gallery">Gallery</MobileNavLink>
            <MobileNavLink target="services">Services</MobileNavLink>
            <MobileNavLink target="/for-beginners" isPage>Beginners Hub</MobileNavLink>
            <MobileNavLink target="community">Community</MobileNavLink>
            <button 
              onClick={() => handleNavClick('contact')} 
              className="block w-full bg-primary-blue text-white px-6 py-3 rounded-lg text-center font-medium mt-4"
            >
              Get Started
            </button>
          </div>
        </div>
      )}
    </nav>
  );
}

// client/src/App.tsx
import { Switch, Route } from "wouter";
import { queryClient } from "./lib/queryClient";
import { QueryClientProvider } from "@tanstack/react-query";
import { Toaster } from "@/components/ui/toaster";
import { TooltipProvider } from "@/components/ui/tooltip";
import { useEffect } from "react";

// Existing Pages
import Home from "@/pages/home";
import NotFound from "@/pages/not-found";

// Beginner Hub Pages
import BeginnerHub from "@/pages/beginners/hub";
import StrategyFramework from "@/pages/beginners/strategy";
import DealScorecard from "@/pages/beginners/scorecard";
import UnderwritingChecklist from "@/pages/beginners/checklist";
import ToolkitLanding from "@/pages/beginners/toolkit-landing";
import ToolkitThankYou from "@/pages/beginners/toolkit-thank-you";

function ScrollToTop() {
  useEffect(() => {
    window.scrollTo(0, 0);
  });
  return null;
}

function Router() {
  return (
    <>
      <ScrollToTop />
      <Switch>
        {/* Core Routes */}
        <Route path="/" component={Home}/>
        
        {/* Beginner Hub Routes */}
        <Route path="/for-beginners" component={BeginnerHub} />
        <Route path="/for-beginners/strategy-framework" component={StrategyFramework} />
        <Route path="/for-beginners/deal-scorecard" component={DealScorecard} />
        <Route path="/for-beginners/underwriting-checklist" component={UnderwritingChecklist} />
        
        {/* Lead Magnet Routes */}
        <Route path="/beginner-investor-toolkit" component={ToolkitLanding} />
        <Route path="/thank-you-toolkit" component={ToolkitThankYou} />
        
        {/* Fallback */}
        <Route component={NotFound} />
      </Switch>
    </>
  );
}

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <TooltipProvider>
        <Toaster />
        <Router />
      </TooltipProvider>
    </QueryClientProvider>
  );
}

export default App;
